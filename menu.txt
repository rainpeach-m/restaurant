<template>

    <div class="overlay">
      <h1 class="title">菜单分类</h1>
      <div class="category-grid">
        <div
          v-for="category in categories"
          :key="category.name"
          class="category-card"

        >
          <span>{{ category.name }}</span>
        </div>
      </div>
    </div>
        <!-- 新增中式窗框装饰 -->
    <div class="window-frame left"></div>
    <div class="window-frame right"></div>
    
    <div class="overlay">
      <!-- 标题增加传统印章元素 -->

          <div 
            v-for="category in categories"
            :key="category.name"
            class="category-card"
            @click="goToCategory(category.route)"
            @mouseenter="playRipple"
          >
            <!-- 增加分类图标和动态水波纹 -->
            <div class="card-content">
              <span class="category-icon">{{ getIcon(category.name) }}</span>
              <span class="category-name">{{ category.name }}</span>
            </div>

          </div>
        </div>
      </div>

<script>
export default {
  name: 'Menu',
  data() {
    return {
      categories: [
        { name: '热菜', route: 'hot' },
        { name: '凉菜', route: 'cold' },
        { name: '主食', route: 'main' },
        // { name: '饮品', route: 'drink' },
        { name: '小吃', route: 'snack' }
      ]
    }
  },
  methods: {
    goToCategory(routeName) {
      this.$router.push(`/menu/${routeName}`)
    },
    getIcon(name) {
      const icons = {
        '热菜': '🔥',
        '凉菜': '❄️',
        '主食': '🍚',
        '小吃': '🥟'
      }
      return icons[name] || '✨'
    },
    playRipple(e) {
      const ripple = e.currentTarget.querySelector('.ripple-effect')
      ripple.style.animation = 'none'
      void ripple.offsetHeight // 触发重绘
      ripple.style.animation = null
    }
  }
}
</script>
